SHOW DATABASES;

create database if not exists SMS
default character set gb2312
default collate gb2312_chinese_ci ;

use SMS;

create table if not exists class(
    classname varchar(10) primary key comment '班级名称',
    department varchar(10) not null comment '所属学院',
    counselor char(8) comment '辅导员',
    studyroom char(8) comment '自习室'
);

create table if not exists student(
    sno char(6) primary key comment '学号',
    sname char(8) not null comment '姓名',
    gender char(2) comment '性别',
    age tinyint comment '年龄，默认值为 20',
    classname varchar(10) comment '班级名称，外键关联到class 表的班级名称',
    nativeplace varchar(20) comment '籍贯'
);

create table if not exists course(
    cno char(3) primary key comment '课程号',
    cname varchar(20) not null comment '课程名',
    pre_course char(3) comment '先修课程',
    type enum('选修','公共必修','专业必修') comment '课程性质，枚举类型',
    credit tinyint(10) comment '学分'
);

create table if not exists score(
    sno char(6) primary key comment '学号',
    cno char(3) primary key comment '课程号',
    grade smallint comment '成绩'
);
